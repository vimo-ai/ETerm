pub struct TerminalRenderer<'a> {
    pool: &'a RioTerminalPool,
    layout: TerminalLayout,
    snapshot: TerminalSnapshot,
}

impl<'a> TerminalRenderer<'a> {
    pub fn render_content(&mut self, lines: usize, cols: usize) -> Result<(), RenderError> {
        let rows_data: Vec<RowRenderData> = (0..lines)
            .into_par_iter()
            .map(|row_index| self.process_row(row_index))
            .collect();

        Ok(())
    }
}
